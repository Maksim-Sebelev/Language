Tree    ::= Assign '$'
Assign  ::= Var '=' AddSub ';'
AddSub  ::= MulDiv { ['+' '-'] MulDiv } *
MulDiv  ::= Pow { ['*' '/'] Pow } *
Pow     ::= Func { '^' Func } *
Func    ::= Minus | Bracket | { Ln | Sin | .. | Arccos }

Minus   ::= '-' MulDiv

Ln      :: 'ln'      '(' AddSub ')'
Sin     :: 'sin'     '(' AddSub ')'
Cos     :: 'cos'     '(' AddSub ')'
...
Arccos  ::= 'arccos' '(' AddSub ')'

Bracket ::= '(' Func ')' | Var | Number
Var     ::= 'x'
Nunmber ::= ['0' - '9']+
